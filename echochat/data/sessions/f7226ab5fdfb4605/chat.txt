// Motor setup
AF_DCMotor leftMotor(4, MOTOR12_64KHZ);  // Left motor on M2
AF_DCMotor rightMotor(1, MOTOR12_64KHZ); // Right motor on M1

// Sensor pins
const int sensors[] = {A0, A1, A2, A3, A4}; // IR1 to IR5

// Configuration
int BASE_SPEED = 150;       // Base speed for straight movement
int TURN_SPEED = 200;       // Speed during turns
int LINE_LOSS_SPEED = 200;  // Speed for line recovery

void setup() {
  Serial.begin(9600);
  
  // Initialize motors
  leftMotor.setSpeed(BASE_SPEED);
  rightMotor.setSpeed(BASE_SPEED);
  leftMotor.run(RELEASE);
  rightMotor.run(RELEASE);
}

void loop() {
  // Read all sensors and store their binary values (0 for black, 1 for white)
  int sensorReadings[5];
  for (int i = 0; i < 5; i++) {
    sensorReadings[i] = analogRead(sensors[i]) > 500 ? 1 : 0; // Threshold set to 500
    Serial.print(sensorReadings[i]); // Debugging: Print sensor values
    Serial.print(" ");
  }
  Serial.println();

  // Main control logic based on sensor readings
  if (sensorReadings[2] == 0) {              // Middle sensor (IR3) detects black line
    straight();
  } else if (sensorReadings[0] == 0) {       // Far left (IR1) detects black line
    sharpLeft();
  } else if (sensorReadings[1] == 0) {       // Left (IR2) detects black line
    leftTurn();
  } else if (sensorReadings[3] == 0) {       // Right (IR4) detects black line
    rightTurn();
  } else if (sensorReadings[4] == 0) {       // Far right (IR5) detects black line
    sharpRight();
  } else {                                   // All sensors detect white surface (line lost)
    lineLost();
  }

  delay(10);
}

// Movement functions
void straight() {
  leftMotor.setSpeed(BASE_SPEED);
  rightMotor.setSpeed(BASE_SPEED);
  bothForward();
}

void leftTurn() {
  leftMotor.setSpeed(BASE_SPEED * 0.6);
  rightMotor.setSpeed(TURN_SPEED);
  bothForward();
}

void sharpLeft() {
  leftMotor.setSpeed(BASE_SPEED * 0.3);
  rightMotor.setSpeed(TURN_SPEED);
  bothForward();
}

void rightTurn() {
  leftMotor.setSpeed(TURN_SPEED);
  rightMotor.setSpeed(BASE_SPEED * 0.6);
  bothForward();
}

void sharpRight() {
  leftMotor.setSpeed(TURN_SPEED);
  rightMotor.setSpeed(BASE_SPEED * 0.3);
  bothForward();
}

void lineLost() {
  // Rotate clockwise to find the line again
  leftMotor.setSpeed(LINE_LOSS_SPEED);
  rightMotor.setSpeed(LINE_LOSS_SPEED);
  
  leftMotor.run(FORWARD);
  rightMotor.run(BACKWARD);

  delay(50); // Short rotation burst to search for the line
}

void bothForward() {
  leftMotor.run(FORWARD);
  rightMotor.run(FORWARD);
}

#include <LiquidCrystal_I2C.h>

// LCD config: 0x27 is the common I2C address. Use 0x3F if 0x27 doesnâ€™t work.
LiquidCrystal_I2C lcd(0x27, 16, 2);

// IR Sensor Pins
const int L2 = A0;
const int L1 = A1;
const int C  = A2;
const int R1 = A3;
const int R2 = 12; // Using D12 for R2

// Motor Pins (connected to L298N)
const int IN1 = 8;
const int IN2 = 9;
const int IN3 = 10;
const int IN4 = 11;

void setup() {
  // LCD setup
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("Line Follower");

  // IR sensors
  pinMode(L2, INPUT);
  pinMode(L1, INPUT);
  pinMode(C, INPUT);
  pinMode(R1, INPUT);
  pinMode(R2, INPUT);

  // Motor pins
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
}

void loop() {
  int l2 = digitalRead(L2);
  int l1 = digitalRead(L1);
  int c  = digitalRead(C);
  int r1 = digitalRead(R1);
  int r2 = digitalRead(R2);

  // Forward
  if (l1 == 1 && c == 0 && r1 == 1) {
    moveForward();
    showLCD("Moving Forward");
  }
  // Slight Left
  else if (l1 == 0 && c == 0) {
    slightLeft();
    showLCD("Slight Left");
  }
  // Sharp Left
  else if (l2 == 0) {
    turnLeft();
    showLCD("Sharp Left");
  }
  // Slight Right
  else if (r1 == 0 && c == 0) {
    slightRight();
    showLCD("Slight Right");
  }
  // Sharp Right
  else if (r2 == 0) {
    turnRight();
    showLCD("Sharp Right");
  }
  // Stop if no line detected
  else {
    stopMotors();
    showLCD("Stopped / Lost");
  }
}

// -------------------- Motor Control Functions --------------------

void moveForward() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void slightLeft() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void slightRight() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
}

void turnLeft() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

def extract_entries(line):
    pattern = r'([A-Z0-9]+)\s+(I(?:-Non)?|VII|II)?\s*([0-9]+)\s+\(([\d.]+)\)'
    matches = re.findall(pattern, line)
    result = {}
    for cat, round_name, rank, percentile in matches:
        round_name = round_name if round_name else "I"
        result[cat] = {
            "round": round_name,
            "rank": int(rank),
            "percentile": float(percentile)
        }
    return result

def parse_cutoff_pdf(pdf_path):
    reader = PdfReader(pdf_path)
    structured_data = defaultdict(lambda: defaultdict(lambda: defaultdict(dict)))

    college = branch = seat_type = None

    for page in reader.pages:
        lines = page.extract_text().split("\n")
        for line in lines:
            college_match = re.match(r"^\d{5} - (.+)", line)
            if college_match:
                college = college_match.group(1).strip()
                continue

            branch_match = re.match(r"^\d{10} - (.+)", line)
            if branch_match:
                branch = branch_match.group(1).strip()
                continue

            seat_type_match = re.match(
                r"^(State Level|Home University Seats Allotted.*|Other Than Home University.*|Minority Seats Allotted.*)",
                line
            )
            if seat_type_match:
                seat_type = seat_type_match.group(1).strip()
                continue

            if college and branch and seat_type:
                entry = extract_entries(line)
                for category, data in entry.items():
                    structured_data[college][branch][seat_type][category] = data

    return structured_data

def save_json(data, output_path):

This is a classic case where an integrated **Project Management Application (PMA)**, especially one leveraging AI/ML, can bring immense value.

Here's a comprehensive approach to solve this problem, broken down into what an AI/ML-powered Project Management Application would do:

## Solving the Project Management Problem with an AI/ML Application

The core solution is to develop and implement an **integrated digital platform** that centralizes project data, automates tasks, and provides intelligent insights.

### 1. Centralized Data Management & Integration

* **Problem Addressed:** "Scattered tools and manual methods."
* **Solution:**
    * **Single Source of Truth:** Create a central database for all project-related information: tasks, timelines, resources, budgets, risks, issues, documents, communication logs.
    * **Tool Integration:** Develop APIs or connectors to integrate with existing essential tools (e.g., communication platforms like Slack/Teams, code repositories like Git, CRM systems, accounting software). This avoids manual data transfer and ensures data consistency.
    * **Standardized Data Entry:** Implement structured forms and templates for task creation, progress updates, and resource allocation to ensure consistent data quality.

### 2. Enhanced Visibility & Reporting (Addressing "Poor Visibility")

* **Problem Addressed:** "Poor visibility."
* **Solution:**
    * **Interactive Dashboards:** Create real-time dashboards that provide a holistic view of all projects or specific projects.
        * **Project Health:** RAG (Red/Amber/Green) status, completion percentage.
        * **Timeline View:** Gantt charts, Kanban boards, or calendar views showing progress against schedule.
        * **Resource Allocation:** Who is working on what, their current load.
        * **Budget vs. Actuals:** Financial tracking.
    * **Customizable Reports:** Allow users to generate on-demand reports on various aspects: task completion, team performance, budget burn, risk status.
    * **Automated Reporting:** Schedule and send regular summary reports to stakeholders.

### 3. Intelligent Tracking & Progress Monitoring (Addressing "Tracking Challenges")

* **Problem Addressed:** "Tracking challenges."
* **Solution:**
    * **Automated Progress Updates:** Integrate with tools (e.g., code commits, file updates, time tracking) to automatically update task progress where possible.
    * **AI-Powered Anomaly Detection:**
        * **Flagging Deviations:** An AI model can analyze historical project data and current progress to identify tasks or projects that are deviating from their planned path (e.g., taking too long, consuming too many resources, not hitting milestones).
        * **Early Warning System:** Proactively alert project managers to potential issues *before* they become critical problems.
    * **Dependency Management:** Automatically identify and visualize task dependencies. Alert if a preceding task is delayed, impacting subsequent tasks.

### 4. Predictive Scheduling & Risk Mitigation (Addressing "Missed Deadlines")

* **Problem Addressed:** "Missed deadlines."
* **AI/ML Solution:**
    * **Predictive Analytics for Timelines:** Based on historical data (similar tasks, similar team members, past performance), AI can predict the most likely completion date for tasks and projects, offering more realistic estimates.
    * **Risk Prediction:** AI can identify patterns in past projects that led to delays or failures (e.g., specific task types, resource conflicts, external dependencies) and flag similar patterns in current projects as potential risks.
    * **Scenario Planning:** Allow project managers to simulate "what-if" scenarios (e.g., "What if this task is delayed by 3 days?") and see the predicted impact on the overall timeline.
    * **Proactive Recommendations:** The system can suggest actions to mitigate predicted risks or recover from potential delays (e.g., reallocating resources, adjusting task priorities).

### 5. Optimized Resource Allocation & Utilization (Addressing "Inefficient Resource Utilization")

* **Problem Addressed:** "Inefficient resource utilization."
* **AI/ML Solution:**
    * **Resource Skill Mapping:** Maintain a database of team members' skills, availability, and workload.
    * **Intelligent Resource Allocation:** An AI algorithm can suggest the optimal team members for specific tasks based on their skills, current workload, and project priorities, preventing both over-allocation and under-utilization.
    * **Workload Balancing:** Identify bottlenecks where a few resources are overloaded while others are underutilized, and suggest re-balancing.
    * **Capacity Planning:** Provide insights into future resource needs based on the project pipeline, helping management plan hiring or training.
    * **Performance Analytics:** Analyze individual and team performance over time to identify areas for improvement or training needs.

### Example Components of an AI/ML Project Management Application:

* **Frontend (User Interface):** Web-based application (React, Angular, Vue.js), potentially mobile apps.
* **Backend (Application Logic):** Python (with Flask/Django), Node.js, Java, .NET.
* **Database:** Relational (PostgreSQL, MySQL) for structured project data, potentially NoSQL (MongoDB) for flexible task details or unstructured notes.
* **AI/ML Core:** Python with libraries like:
    * **Scikit-learn:** For predictive modeling (regression, classification), clustering (for resource grouping).
    * **TensorFlow/PyTorch:** For more complex deep learning models if needed (e.g., natural language processing for analyzing project communications or risk descriptions).
    * **Pandas/NumPy:** For data manipulation and analysis.
* **Data Visualization:** Matplotlib, Seaborn (for internal analytics), or integration with business intelligence tools like Tableau/Power BI.
* **Cloud Platform:** AWS, Azure, GCP for scalability, managed services (databases, ML services).
* **Version Control:** Git/GitHub for code management.

By building an application with these features, you transform project management from a chaotic, manual process into a streamlined, insight-driven, and proactive operation.
